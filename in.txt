class Main {
    static string[] bar(int opt) {
        if {
            opt == 1:
                return [1];
        ||| opt == 2: {
                var arr1 = [2];
                var arr2 = [3];
                return arr1 %% 2 ++ arr2;
            }
        }
    }

    static void main() {
        foreach (var s in bar(2)) {
            Print(s);
        }

        foreach (var s in bar(1)) {
            Print(s);
        }
    }
}
